```
frontend/
├── src/
│   ├── main.tsx              # 入口
│   ├── App.tsx               # 根组件
│   ├── routes.tsx            # 路由配置
│   ├── components/           # 公共组件
│   ├── pages/                # 页面组件
│   │   ├── Home/
│   │   └── Admin/
│   ├── stores/               # Zustand stores
│   ├── utils/                # 工具函数
│   └── types/                # TypeScript 类型
├── public/                   # 静态资源
├── index.html
├── vite.config.ts
├── tsconfig.json
├── package.json
└── .env
```
